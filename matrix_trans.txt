def transpose(matrix):

    result = [[matrix[j][i] for j in range(len(matrix))] for i in range(len(matrix[0]))]

    return result


print(transpose([[1, 2], [3, 4], [5, 6]])) 

# [[1, 3, 5], [2, 4, 6]]

print(transpose([[1, 2], [3, 4], [5, 6], [7, 8]]))

# [[1, 3, 5, 7], [2, 4, 6, 8]]
