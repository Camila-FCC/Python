def make_leet(s):

    return s.replace("a", "4").replace("e", "3").replace("g", "9").replace("i", "1").replace("l", "1").replace("o", "0").replace("s", "5").replace("t", "7")

print(make_leet("abcdefghijklmnopqrstuvwxyz"))
