def get_hill_rating(drop, distance, hill_type):

    downhill = 1.2 if hill_type == "Downhill" else 0

    slalom = 0.9 if hill_type == "Slalom" else 0

    giant_slalom = 1.0 if hill_type == "Giant Slalom" else 0

    steepness = drop/distance

    adj_steepness = steepness * (downhill if hill_type == "Downhill" else slalom if hill_type == "Slalom" else giant_slalom if hill_type == "Giant Slalom" else 1)

    return "Green" if adj_steepness <= 0.1 else "Blue" if adj_steepness <= 0.25 else "Black" if adj_steepness > 0.25 else ""


print(get_hill_rating(620, 2800, "Downhill"))
# "Black"
