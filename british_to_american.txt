def british_to_american(sentence):

    sentence = sentence.replace("colour", "color").replace("Colour", "Color").replace("flavour", "flavor").replace("Flavour", "Flavor").replace("honour", "honor").replace("Honour", "Honor").replace("neighbour", "neighbor").replace("Neighbour", "Neighbor").replace("labour", "labor").replace("Labour", "Labor").replace("humour", "humor").replace("Humour", "Humor").replace("centre", "center").replace("Centre", "Center").replace("fibre", "fiber").replace("Fibre", "Fiber").replace("defence", "defense").replace("Defence", "Defense").replace("offence", "offense").replace("Offence", "Offense").replace("organise", "organize").replace("Organise", "Organize").replace("Recognise", "Recognize").replace("recognise", "recognize").replace("analyse", "analyze").replace("Analyse", "Analyze")

    return sentence


print(british_to_american("I love the colour blue."))

# "I love the color blue."

# "The fiber optic cable is new."

# "It's an honor to meet someone with such humor."

print(british_to_american("The unrecognised artist analysed his colour palette at the centre."))

# "The unrecognized artist analyzed his color palette at the center."
