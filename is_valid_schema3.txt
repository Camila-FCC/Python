def is_valid_schema(obj):

    if len(list(obj.keys())) < 4:
        return False


    roles = ["user", "creator", "moderator", "staff", "admin"]
    
    obj_keys = list(obj.keys())

    username_key = isinstance(obj["username"], str)

    posts_key = isinstance(obj["posts"], int)

    verified_key = isinstance(obj["verified"], bool)

    role = True if obj["role"] in roles else False 

    #print(role)

    check_all = username_key and posts_key and verified_key and role

    return check_all
