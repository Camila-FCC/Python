def avalanche_risk(snow_depth, slope):

    if snow_depth == "Shallow":
        return "Safe"

    if snow_depth == "Moderate" or snow_depth == "Deep":
        return "Safe" if slope == "Gentle" else "Risky"


print(avalanche_risk("Shallow", "Very Steep"))
# "Safe"
