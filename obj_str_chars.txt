def get_frequency(s):

    s = list(s)
    #print(s)

    obj_str = {}

    i = 0

    while i < len(s):

        obj_str[f'{s[i]}'] = len(list(filter(lambda x : x == s[i], s)))

        i+=1

    return obj_str


print(get_frequency("She sells seashells by the seashore."))
