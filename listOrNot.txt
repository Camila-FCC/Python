def is_flat(arr):

    return all(not isinstance(item, list) for item in arr)

print(is_flat([1, 2, 3, 4]))
# True

print(is_flat([1, [2, 3], 4]))
# False

print(is_flat([1, 0, False, True, "a", "b"]))
# True

print(is_flat(["a", [0], "b", True]))
# False

print(is_flat([1, [2, [3, [4, [5]]]], 6]))
# False
