def convert_words(s):

    s = s.split(" ")

    word_length = list(map(lambda w: str(len(w)), s))

    return " ".join(word_length)

convert_words("Lorem ipsum dolor sit amet consectetur adipiscing elit donec ut ligula vehicula iaculis orci vel semper nisl"))
