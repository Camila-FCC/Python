def convert_to_km(miles):

    one_mile_to_km = 1.60934

    if(miles < 0):
        return print("Only positive numbers, please.")
    
    # 1 mile = 1.60934 km
    #  miles =   ? km

    calc = round((one_mile_to_km * miles), 2)

    return calc
